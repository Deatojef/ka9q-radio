204// Fast approximate square root, for signal magnitudes
205// https://dspguru.com/dsp/tricks/magnitude-estimator/
206static inline float approx_magf(complex float x){
207 static float const Alpha = 0.947543636291;
208 static float const Beta = 0.392485425092;
209
210 float absr = fabsf(__real__ x);
211 float absi = fabsf(__imag__ x);
212
213 return Alpha * max(absr,absi) + Beta * min(absr,absi);
214}

234 // Will this long conditional help the optimizer?
235 return (a->tv_sec > b->tv_sec) ? 1

245// Convert timespec (seconds, nanoseconds) to integer nanoseconds
246// Integer nanoseconds overflows past 584.94242 years. That's probably long enough
247static inline long long ts2ns(struct timespec const *ts){
248 return ts->tv_sec * BILLION + ts->tv_nsec;
249}
250// Convert integer nanosec count to timspec
251static inline void ns2ts(struct timespec *ts,long long ns){
252 lldiv_t r = lldiv(ns,BILLION);
253 ts->tv_sec = r.quot;
254 ts->tv_nsec = r.rem;
255}

290// Wrap pointer p to keep it in range (base, base + size), where size is in bytes
291// The callers use C casts in a somewhat dodgy fashion, but is OK because size is always a multiple of the page size,
292// and there's an integral number of the objects we're pointing to in a page (we hope!!)
293static inline void mirror_wrap(void const **p, void const * const base,size_t const size){
294 assert(*p >= base); // Shouldn't be THIS low
295 assert(*p < base + 2 * size); // Or this high
296
297 if((uint8_t *)*p >= (uint8_t *)base + size)
298 *p = (uint8_t *)*p - size;
299}
