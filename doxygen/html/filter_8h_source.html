91// Write complex sample to input side of filter
92static inline int put_cfilter(struct filter_in * restrict const f,complex float const s){ // Complex
93 assert((void *)(f->input_write_pointer.c) >= f->input_buffer);
94 assert((void *)(f->input_write_pointer.c) < f->input_buffer + f->input_buffer_size);
95 *f->input_write_pointer.c++ = s;
96 mirror_wrap((void *)&f->input_write_pointer.c, f->input_buffer,f->input_buffer_size);
97 if(++f->wcnt >= f->ilen){
98 f->wcnt -= f->ilen;
99 execute_filter_input(f);
100 return 1; // may now execute filter output without blocking
101 }
102 return 0;
103}
